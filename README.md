created by pedro ferrusca

# create virtual env (use python 3.11 at least)
`
    python -m venv env
    source bin/env/activate # mac or linux
`

# Installation

`
    pip install -r requirements.txt
`

# Usage UI APP
`
    python -m socket_app.app
`

# Usage chat server
`
    python -m socket_app.chat_server
`
